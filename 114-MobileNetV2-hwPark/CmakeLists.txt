cmake_minimum_required(VERSION 3.16)

project(114-MobileNetV2-hwPark)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)

get_filename_component(REPO_ROOT ${CMAKE_CURRENT_SOURCE_DIR}/.. ABSOLUTE)

find_package(Vulkan REQUIRED)

if(NOT TARGET nlohmann_json::nlohmann_json)
    if(EXISTS ${REPO_ROOT}/external/json/CMakeLists.txt)
        add_subdirectory(${REPO_ROOT}/external/json ${CMAKE_BINARY_DIR}/external/json)
    else()
        find_package(nlohmann_json CONFIG REQUIRED)
    endif()
endif()

if(NOT TARGET spirv-reflect-static AND EXISTS ${REPO_ROOT}/external/SPIRV-Reflect/CMakeLists.txt)
    set(SPIRV_REFLECT_STATIC_LIB ON CACHE BOOL "Build a SPIRV-Reflect static library" FORCE)
    add_subdirectory(${REPO_ROOT}/external/SPIRV-Reflect ${CMAKE_BINARY_DIR}/external/SPIRV-Reflect)
endif()

add_executable(${PROJECT_NAME}
    main.cpp
    core/vulkanApp.cpp
    nodes/base_node.cpp
    nodes/neural_nodes.cpp
    model/mobilenet_v2.cpp
    dataloader/image_loader.cpp
    utils/jsonParser.cpp
    utils/spirvHelpers.cpp
)


add_definitions(
    -DPROJECT_CURRENT_DIR="${CMAKE_CURRENT_SOURCE_DIR}"
)

target_link_libraries(${PROJECT_NAME} PRIVATE
    glfw
    spirv-reflect-static
    Vulkan::Vulkan
    glslang
    glslang-default-resource-limits
    nlohmann_json::nlohmann_json
)

target_include_directories(${PROJECT_NAME} PRIVATE 
    ${CMAKE_CURRENT_SOURCE_DIR}
)
